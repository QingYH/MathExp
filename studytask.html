MATLAB代码示例 - 欧拉法解微分方程
% MATLAB欧拉法解微分方程
f = @(t,y) -2*y + t; % dy/dt = -2y + t
t0 = 0; y0 = 1;      % 初始条件
h = 0.1;             % 步长
t = t0:h:2;          % 时间区间
y = zeros(size(t));
y(1) = y0;

for i = 1:length(t)-1
    y(i+1) = y(i) + h*f(t(i), y(i));
end

plot(t,y,'-o');
title('Euler Method Solution');

% MATLAB方波傅里叶级数合成
t = linspace(-pi, pi, 1000);
f = zeros(size(t));
N = 10; % 谐波次数

for n = 1:2:N  % 只取奇数次谐波
    f = f + (4/pi) * sin(n*t)/n;
end

plot(t, f, 'LineWidth', 2);
hold on;
plot(t, sign(t), '--'); % 理想方波
title(['傅里叶级数逼近 (N=', num2str(N), ')']);
legend('合成波形', '理想方波');
xlabel('时间'); ylabel('幅值');
grid on;
Mathematica代码示例 - 傅里叶系数计算
(* Mathematica傅里叶系数计算 *)
f[x_] := Piecewise[{{1, 0 <= x < Pi}, {-1, -Pi <= x < 0}}]
a[n_] := (1/Pi) Integrate[f[x] Cos[n x], {x, -Pi, Pi}]
b[n_] := (1/Pi) Integrate[f[x] Sin[n x], {x, -Pi, Pi}]
FourierSeries = Table[
   a[0]/2 + Sum[a[k] Cos[k x] + b[k] Sin[k x], {k, 1, n}], 
   {n, 1, 5}];
Manipulate[
 Plot[{f[x], FourierSeries[[n]]}, {x, -Pi, Pi}, 
  PlotRange -> {-1.5, 1.5}, 
  PlotLegends -> {"原函数", "傅里叶逼近"}], 
 {n, 1, 5, 1}]
Python代码示例 - 傅里叶频谱分析
# Python傅里叶频谱分析
import numpy as np
import matplotlib.pyplot as plt

def square_wave(t):
    return np.sign(np.sin(t))

t = np.linspace(-np.pi, np.pi, 1000)
f = square_wave(t)

# 计算FFT
fft = np.fft.fft(f)
freq = np.fft.fftfreq(len(t), t[1]-t[0])

plt.figure(figsize=(12,4))
plt.subplot(121)
plt.plot(t, f)
plt.title('时域信号')

plt.subplot(122)
plt.stem(freq[:20], np.abs(fft[:20]), 'r')
plt.title('频域频谱')
plt.tight_layout()
